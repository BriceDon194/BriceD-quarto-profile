Simple Line Chart

Letâ€™s create a simple line chart showing some sample data:

import matplotlib.pyplot as plt
import numpy as np

# Create sample data
x = np.linspace(0, 10, 100)
y = np.sin(x)

# Create the plot
plt.figure(figsize=(7, 4.75))
plt.plot(x, y, 'b-', linewidth=2, label='sin(x)')
plt.xlabel('X values')
plt.ylabel('Y values')
plt.title('My First Matplotlib Chart')
plt.grid(True, alpha=0.3)
plt.legend()
plt.show()

Bar Chart Example

Hereâ€™s a bar chart showing some sample data:

# Sample data for bar chart
categories = ['A', 'B', 'C', 'D', 'E']
values = [23, 45, 56, 78, 32]

plt.figure(figsize=(7, 4.75))
bars = plt.bar(categories, values, color=['red', 'green', 'blue', 'orange', 'purple'])
plt.xlabel('Categories')
plt.ylabel('Values')
plt.title('Sample Bar Chart')

# Add value labels on bars
for bar, value in zip(bars, values):
    plt.text(bar.get_x() + bar.get_width()/2, bar.get_height() + 1, 
             str(value), ha='center', va='bottom')

plt.show()

    ['Nissan Leaf', 147, 111],    # MPGe
    ['Chevrolet Bolt', 200, 120],  # MPGe
]

# Extract data
car_names = [car[0] for car in cars_data]
horsepower = [car[1] for car in cars_data]
mpg = [car[2] for car in cars_data]

# Create scatter plot
plt.figure(figsize=(10, 6))
scatter = plt.scatter(horsepower, mpg, alpha=0.7, s=100, c=horsepower, 
                     cmap='viridis', edgecolors='black', linewidth=0.5)

# Add labels for some notable cars
notable_cars = ['Toyota Prius', 'Tesla Model 3', 'Chevrolet Corvette', 
                'Ford F-150', 'Honda Civic', 'Porsche 911']
for i, name in enumerate(car_names):
    if name in notable_cars:
        plt.annotate(name, (horsepower[i], mpg[i]), 
                    xytext=(5, 5), textcoords='offset points', 
                    fontsize=8, alpha=0.8)

plt.xlabel('Horsepower (HP)', fontsize=11)
plt.ylabel('Fuel Efficiency (MPG / MPGe)', fontsize=11)
plt.title('Car Performance: Horsepower vs Fuel Efficiency\n(Real Car Data)', fontsize=12)
plt.colorbar(scatter, label='Horsepower (HP)')
plt.grid(True, alpha=0.3)
plt.tight_layout()
plt.show()
